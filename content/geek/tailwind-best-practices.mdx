## 示例代码

```tsx
// tailwind.config.ts
import type { Config } from "tailwindcss"

const config: Config = {
  content: [
    "./components/**/*.{ts,tsx}",
    "./app/**/*.{ts,tsx}",
  ],
  theme: {
    extend: {
      colors: {
        brand: {
          50: "#f0f9ff",
          900: "#0c4a6e",
        },
      },
    },
  },
  plugins: [],
}

export default config
```
